222  public void setup(List<?> localVariableNodes, InsnList instructions) {
223  variables = new LocalVariables(localVariableNodes, instructions);
224  originalInsnListSize = instructions.size();
225  for (int i = 0; i < instructions.size(); ++i) {
226  AbstractInsnNode node = instructions.get(i);
227 
228  if (node.getType() == AbstractInsnNode.LABEL) {
229  // Record label location
230  Label label = ((LabelNode) node).getLabel();
231  labelInstructionIndexMap.put(label, i);
232 
233  } else if (node.getType() == AbstractInsnNode.LINE) {
234  // Record line numbers corresponding to labels (because LINE is always placed AFTER its LABEL)
235  LineNumberNode line = (LineNumberNode)node;
236  LabelNode label = line.start;
237  labelLineNumberMap.put(label.getLabel(), line.line);
238 
239  // If the line number node points to the first instruction of the method,
240  // we regard the line as the method declaration line
241  if (label == instructions.getFirst()) {
242  currentLine = line.line;
243  }
244  }
245  }
246 
247  String hash = "";
248  try {
249  MessageDigest digest = MessageDigest.getInstance("SHA-1");
250  for (int i = 0; i < instructions.size(); ++i) {
251  String line = getInstructionString(instructions, i) + "\n";
252  digest.update(line.getBytes());
253  }
254  StringBuilder buf = new StringBuilder();
255  for (byte b: digest.digest()) {
256  int c = b;
257  c &= 0xff;
258  buf.append(Character.forDigit(c / 16, 16));
259  buf.append(Character.forDigit(c % 16, 16));
260  }
261  hash = buf.toString();
262  } catch (NoSuchAlgorithmException e) {
263  }
264 
265  weavingInfo.startMethod(className, methodName, methodDesc, access, sourceFileName, hash);
266  weavingInfo.nextDataId(currentLine, -1, EventType.RESERVED, Descriptor.Void, null);

575  @Override
576  public void visitMethodInsn(int opcode, String owner, String name, String desc, boolean itf) {
577 
578  // This method call is a constructor chain if the call initializes
579  // "this" object by this() or suepr().
580  boolean isConstructorChain = name.equals("<init>") && methodName.equals("<init>")
581  && newInstructionStack.isEmpty();
582  assert !isConstructorChain || opcode == Opcodes.INVOKESPECIAL : "A constructor chain must use INVOKESPECIAL.";
583 
584  // Pop a corresponding new instruction if this method call initializes
585  // an object.
586  ANewInstruction newInstruction = null;
587  if (!isConstructorChain && name.equals("<init>")) {
588  newInstruction = newInstructionStack.pop();
589  assert newInstruction.getTypeName().equals(owner);
590  }
591 
592  // Generate instructions to record method call and its parameters
593  if (config.recordMethodCall()) {
594 
595  InstructionAttributes attr = InstructionAttributes.of(ATTRIBUTE_OPCODE, OpcodesUtil.getString(opcode))
596  .and(ATTRIBUTE_OWNER, owner)
597  .and(ATTRIBUTE_NAME, name)
598  .and(ATTRIBUTE_DESCRIPTOR, desc);
599 
600  if (config.recordParameters()) {
601  // Generate code to record parameters
602  MethodParameters params = new MethodParameters(desc);
603 
604  // Store parameters except for a receiver into additional local
605  // variables.
606  for (int i = params.size() - 1; i >= 0; i--) {
607  int local = super.newLocal(params.getType(i));
608  params.setLocalVar(i, local);
609  generateNewVarInsn(params.getStoreInstruction(i), local);
610  }
611  // Here, all parameters (except for a receiver) are stored in
612  // local variables.
613 
614  boolean hasReceiver = (opcode != Opcodes.INVOKESTATIC);
615  boolean receiverNotInitialized = isConstructorChain || (newInstruction != null);
616 
617  // Duplicate an object reference to record the created object
618  int offset;
619  if (receiverNotInitialized) {
620  // For constructor, duplicate the object reference, and record it later.
621  // Here, record only the execution of the call.
622  super.visitInsn(Opcodes.DUP);
623  attr.and(ATTRIBUTE_METHOD_TYPE, "constructor");
624  if (newInstruction != null) {
625  attr.and(ATTRIBUTE_CREATION_LOCATION, newInstruction.getInstructionIndex());
626  }
627  generateLogging(EventType.CALL, Descriptor.Void, attr);
628  offset = 1;
629  } else if (hasReceiver) { // For a regular non-static method,
630  // duplicate and record the object
631  // reference.
632  super.visitInsn(Opcodes.DUP);
633  generateLogging(EventType.CALL, Descriptor.Object, attr.and(ATTRIBUTE_METHOD_TYPE, "instance"));
634  offset = 1;
635  } else { // otherwise, no receivers.
636  generateLogging(EventType.CALL, Descriptor.Void, attr.and(ATTRIBUTE_METHOD_TYPE, "static"));
637  offset = 0;
638  }
639 
640  // Record remaining parameters
641  int paramIndex = 0;
642  while (paramIndex < params.size()) {
643  generateNewVarInsn(params.getLoadInstruction(paramIndex), params.getLocalVar(paramIndex));
644  InstructionAttributes a = InstructionAttributes.of(ATTRIBUTE_INDEX, paramIndex + offset)
645  .and(ATTRIBUTE_TYPE, params.getType(paramIndex).getDescriptor());
646  generateLogging(EventType.CALL_PARAM, params.getRecordDesc(paramIndex), a);
647  paramIndex++;
648  }
649 
650  // Restore parameters from local variables
651  for (int i = 0; i < params.size(); i++) {
652  generateNewVarInsn(params.getLoadInstruction(i), params.getLocalVar(i));
653  }
654 
655  // Store the current location for exceptional exit
656  generateLocationUpdate();
657  // Call the original method
658  super.visitMethodInsn(opcode, owner, name, desc, itf);
659 
660  // record return value
661  String returnDesc = getReturnValueDesc(desc);
662  Descriptor d = Descriptor.get(returnDesc);
663  generateLoggingPreservingStackTop(EventType.CALL_RETURN, d, InstructionAttributes.of(ATTRIBUTE_TYPE, returnDesc));
664 
665  if (isConstructorChain) {
666  if (config.recordExecution()) {
667  // Record an object initialized by this() or super()
668  generateLogging(EventType.METHOD_OBJECT_INITIALIZED, Descriptor.Object, null);
669  }
670  afterInitialization = true;
671  } else if (newInstruction != null) {
672  // Record an object created by "new X()"
673  generateLogging(EventType.NEW_OBJECT_CREATED, Descriptor.Object, InstructionAttributes.of(ATTRIBUTE_CREATION_LOCATION, newInstruction.getInstructionIndex()));
674  }
675 
676 
677  } else { // !recordParameters
678  // Call an occurrence of a call
679  if (newInstruction != null) {
680  attr.and(ATTRIBUTE_CREATION_LOCATION, newInstruction.getInstructionIndex());
681  }
682  generateLogging(EventType.CALL, Descriptor.Void, attr);
683 
684  // Store the current location for exceptional exit
685  generateLocationUpdate();
686 
687  // Call the original method
688  super.visitMethodInsn(opcode, owner, name, desc, itf);
689 
690  // Record return event
691  generateLogging(EventType.CALL_RETURN, Descriptor.Void, null);
692 
693  // Constructor call
694  if (isConstructorChain) {
695  if (config.recordExecution()) {
696  super.visitVarInsn(Opcodes.ALOAD, 0);
697  generateLogging(EventType.METHOD_OBJECT_INITIALIZED, Descriptor.Object, null);
698  }
699  afterInitialization = true;
700  }
701 
702  }
703 
704 
705  } else {
706 
707  super.visitMethodInsn(opcode, owner, name, desc, itf);
708 
709  // Constructor call
710  if (isConstructorChain) {
711  if (config.recordExecution()) {
712  super.visitVarInsn(Opcodes.ALOAD, 0);
713  generateLogging(EventType.METHOD_OBJECT_INITIALIZED, Descriptor.Object, null);
714  }
715  afterInitialization = true;
716  }
717  }
718 
719  // If this call is a constructor-chain (super()/this() at the beginning
720  // of a constructor), start a try block to record an exception thrown by
721  // the remaining code.
722  // Because Java Verifier does not allow "try { super(); } catch ... ",
723  // this code generate "super(); try { ... }".
724  if (isConstructorChain && (config.recordExecution() || config.recordCatch())) {
725  super.visitLabel(startLabel);
726  isStartLabelLocated = true;
727  }
728 
729  instructionIndex++;
