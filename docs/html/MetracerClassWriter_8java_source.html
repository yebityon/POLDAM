35  protected String getCommonSuperClass(String theType1, String theType2) {
36  try {
37  ClassReader info1 = typeInfo(theType1);
38  ClassReader info2 = typeInfo(theType2);
39 
40  if ((info1.getAccess() & Opcodes.ACC_INTERFACE) != 0) {
41  if (typeImplements(theType2, info2, theType1)) {
42  return theType1;
43  }
44  if ((info2.getAccess() & Opcodes.ACC_INTERFACE) != 0) {
45  if (typeImplements(theType1, info1, theType2)) {
46  return theType2;
47  }
48  }
49  return "java/lang/Object";
50  }
51 
52  if ((info2.getAccess() & Opcodes.ACC_INTERFACE) != 0) {
53  if (typeImplements(theType1, info1, theType2)) {
54  return theType2;
55  } else {
56  return "java/lang/Object";
57  }
58  }
59 
60  StringBuilder b1 = typeAncestors(theType1, info1);
61  StringBuilder b2 = typeAncestors(theType2, info2);
62  String result = "java/lang/Object";
63  int end1 = b1.length();
64  int end2 = b2.length();
65 
66  while (true) {
67  int start1 = b1.lastIndexOf(";", end1 - 1);
68  int start2 = b2.lastIndexOf(";", end2 - 1);
69  if (start1 != -1 && start2 != -1
70  && end1 - start1 == end2 - start2) {
71  String p1 = b1.substring(start1 + 1, end1);
72  String p2 = b2.substring(start2 + 1, end2);
73  if (p1.equals(p2)) {
74  result = p1;
75  end1 = start1;
76  end2 = start2;
77  } else {
78  return result;
79  }
80  } else {
81  return result;
82  }
83  }
84  } catch (IOException e) {
85  throw new RuntimeException(e.toString());
86  }
87  }

89  private ClassReader typeInfo(final String theType) throws IOException {
90  StringBuilder visitedLoaders = new StringBuilder();
91  ClassLoader loader = classLoader;
92 
93  while(loader != null) {
94  if(visitedLoaders.length() > 0)
95  visitedLoaders.append(", ");
96 
97  visitedLoaders.append(loader.toString());
98  InputStream is = loader.getResourceAsStream(theType + ".class");
99 
100  if(is != null) {
101  try {
102  return new ClassReader(is);
103  } finally {
104  is.close();
105  }
106  }
107 
108  loader = loader.getParent();
109  }
110 
111  if(visitedLoaders.length() == 0)
112  visitedLoaders.append("<empty>");
113 
114  throw new IOException(String.format("Failed to open %1$s in all known class loaders: %2$s", theType, visitedLoaders.toString()));
115  }

117  private boolean typeImplements(String theType, ClassReader theReader, String theInterface) throws IOException {
118  while (!"java/lang/Object".equals(theType)) {
119  String[] itfs = theReader.getInterfaces();
120 
121  for (int i = 0; i < itfs.length; ++i) {
122  if (itfs[i].equals(theInterface)) {
123  return true;
124  }
125  }
126 
127  for (int i = 0; i < itfs.length; ++i) {
128  if (typeImplements(itfs[i], typeInfo(itfs[i]), theInterface)) {
129  return true;
130  }
131  }
132 
133  theType = theReader.getSuperName();
134  theReader = typeInfo(theType);
135  }
136 
137  return false;
138  }
