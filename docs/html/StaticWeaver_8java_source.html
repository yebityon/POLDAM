181  private void weaveClassesInJarFile(String inputjarName, ZipInputStream inputJar, ZipOutputStream outputJar) throws IOException {
182  for (ZipEntry entry = inputJar.getNextEntry(); entry != null; entry = inputJar.getNextEntry()) {
183  ZipEntry outEntry = new ZipEntry(entry.getName());
184  outputJar.putNextEntry(outEntry);
185 
186  if (entry.getName().endsWith(".class")) {
187  // Weave a class file and write into the output jar
188  byte[] target = streamToByteArray(inputJar, entry.getSize());
189  byte[] result = executeWeave(inputjarName, entry.getName(), target);
190  outputJar.write(result);
191  } else if (entry.getName().endsWith(".jar") && weaveInternalJar) {
192  // Load the internal jar and open it as a stream
193  byte[] internalJar = streamToByteArray(inputJar, entry.getSize());
194  ByteArrayInputStream b = new ByteArrayInputStream(internalJar);
195  ZipInputStream internalJarStream = new ZipInputStream(b);
196 
197  // Create an output stream
198  ByteArrayOutputStream buf = new ByteArrayOutputStream();
199  JarOutputStream bufWriter = new JarOutputStream(buf);
200 
201  weaveClassesInJarFile(inputjarName + "/" + entry.getName(), internalJarStream, bufWriter);
202  bufWriter.close();
203 
204  // Save the result to the parent JAR
205  outputJar.write(buf.toByteArray());
206  } else {
207  // Just copy the content
208  byte[] target = streamToByteArray(inputJar, entry.getSize());
209  outputJar.write(target);
210  }
211  outputJar.closeEntry();
212  }
