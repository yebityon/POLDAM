77POLDAM::OmniGraph buildGraph(POLDAM::poldamConfig config, const std::string inputDir, const std::string outputFileName)
78{
79 std::cout << POLDAM_UTIL::POLDAM_PRINT_SUFFIX << "Reading Metafiles\n";
80 POLDAM::metafileFactory factory(inputDir);
81
82 auto dataids = factory.createInstance<POLDAM::DataIdsParser>("dataids.txt", true);
83 auto seloggerParser = factory.createInstance<POLDAM::SeloggerLogParser>("log-00001.txt");
84 auto objectFileParser = factory.createInstance<POLDAM::ObjectfileParser>();
85 auto methodParser = factory.createInstance<POLDAM::MethodDataParser>("methods.txt", true);
86 auto classesParser = factory.createInstance<POLDAM::ClassesDataParser>("classes.txt", true);
87
88 std::cout
89 << POLDAM_UTIL::POLDAM_PRINT_SUFFIX << "Successfully reading Metafiles\n";
90
91 // Phase2. Create Graph.
92 std::cout << POLDAM_UTIL::POLDAM_PRINT_SUFFIX << "Create Graph\n";
93
94 const std::vector<std::string> omniLog = seloggerParser.getData();
95
96 const std::vector<std::string> dataidsData = dataids.getData();
97 const std::vector<POLDAM::DataId> parsedDataIds = dataids.getParsedData();
98
99 const std::vector<std::string> methodData = methodParser.getData();
100 const std::vector<POLDAM::MethodsData> parsedMethodsData = methodParser.getParsedData();
101
102 const std::vector<std::string> classesData = classesParser.getData();
103 const std::vector<POLDAM::ClassesData> parsedClassesData = classesParser.getParsedData();
104
105 const std::vector<POLDAM::ObjectData> parsedObjectData = objectFileParser.getParsedData();
106
107 std::cout << POLDAM_UTIL::POLDAM_PRINT_SUFFIX << "Successfully Parsed\n";
108 POLDAM::OmniGraph omniGraph{};
109
110 for (const POLDAM::SeloggerData log : seloggerParser.getParserdData())
111 {
112 POLDAM::DataId dataId = parsedDataIds[log.dataId];
113 const POLDAM::MethodsData m = parsedMethodsData[dataId.methodId];
114 const POLDAM::ClassesData c = parsedClassesData[dataId.classId];
115
116 if (dataId.eventType == POLDAM::SELOGGER_EVENT_TYPE::METHOD_ENTRY)
117 {
118 POLDAM::GraphVertex v;
119 const unsigned int classId = dataId.classId;
120
121 v.methodStr = m.methodName;
122 v.methodHash = std::to_string(std::hash<std::string>()(m.className + m.methodName));
123 v.outputFormat = m.methodName;
124
125 bool result = omniGraph.addOmniVertex(v, log.threadId);
126 }
127 else if (dataId.eventType == POLDAM::SELOGGER_EVENT_TYPE::METHOD_PARAM)
128 {
129 const std::string argValue = log.value;
130 }
131 else if (dataId.eventType == POLDAM::SELOGGER_EVENT_TYPE::METHOD_NORMAL_EXIT)
132 {
133 omniGraph.computeHash(log.threadId);
134 omniGraph.popStackVertex(log.threadId);
135 }
136 else if (dataId.eventType == POLDAM::SELOGGER_EVENT_TYPE::CALL_PARAM)
137 {
138 // Use these values to compute param hash.
139 const unsigned int paramIdx = static_cast<unsigned int>(std::stoi(dataId.attr["index"]));
140 const std::string paramType = dataId.attr["type"];
141 std::string paramConcreateValue = "";
142
143 if (paramType.find("String") != std::string::npos)
144 {
145 // fixed for 0-index
146 const int argValueIdx = std::stoi(log.value) - 1;
147 const POLDAM::ObjectData o = parsedObjectData[argValueIdx];
148 omniGraph.updateStackTopVertexParamInfo(o.stringValue, log.threadId);
149 }
150 // branch for paramType that paramValue is directly recored in SELogger.
151 else if (paramType.find("Ljava") == std::string::npos)
152 {
153 // std::cout << "paramType: " << paramType << ",Value: " << log.value << std::endl;
154 omniGraph.updateStackTopVertexParamInfo(log.value, log.threadId);
155 }
156 else
157 {
158 const POLDAM::ObjectData o = parsedObjectData[std::stoi(log.value) - 1];
159 // std::cout << "paramType: " << paramType << ",Value: " << o.objectType << std::endl;
160 }
161 }
162 else
163 {
164 const std::string logString = shapeLogString(dataId, m, c);
165 omniGraph.updateStackTopVertex(logString, log.threadId);
166 }
167 }
168
169 POLDAM::OmniWriter writer(omniGraph);
170 std::cout << POLDAM_UTIL::POLDAM_PRINT_SUFFIX << "writing result..." << std::endl;
171 writer.writeOmniGraph(outputFileName);
172
173 return omniGraph;
174}
